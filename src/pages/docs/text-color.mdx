export const classes = {
  utilities,
  transformProperties: ({ selector, properties }) => {
    delete properties['--tw-text-opacity']
    Object.keys(properties).forEach(key => {
      properties[key] = properties[key].replace(' / var(--tw-text-opacity)', '')
    })
    return properties
  },
  preview: (css, { className }) => (
    <td
      className={`text-center font-medium text-base whitespace-nowrap align-middle ${className}`}
      style={{
        color: Array.isArray(css['color']) ? css['color'][0] : css['color'],
      }}
    >
      <div className="w-16 mx-auto">Aa</div>
    </td>
  ),
}

## Основы использования

### Установка цвета текста

Управляйте цветом текста элемента с помощью утилит `text-{color}`.

```html {{ example: true }}
<div class="relative text-xl text-center font-medium leading-6">
  <p class="text-blue-600 dark:text-sky-400">Быстрая коричневая лиса прыгает через ленивую собаку.</p>
</div>
```

```html
<p class="**text-blue-600**" dark-class="**text-sky-400**">Быстрая коричневая лиса...</p>
```

### Изменение непрозрачности

Управляйте непрозрачностью цвета текста элемента с помощью модификатора непрозрачности цвета.

```html {{ example: true }}
<div class="text-xl font-medium leading-6 space-y-4">
  <p class="text-blue-600/100 dark:text-sky-400/100">Быстрая коричневая лиса прыгает через ленивую собаку.</p>
  <p class="text-blue-600/75 dark:text-sky-400/75">Быстрая коричневая лиса прыгает через ленивую собаку.</p>
  <p class="text-blue-600/50 dark:text-sky-400/50">Быстрая коричневая лиса прыгает через ленивую собаку.</p>
  <p class="text-blue-600/25 dark:text-sky-400/25">Быстрая коричневая лиса прыгает через ленивую собаку.</p>
</div>
```

```html
<p class="text-blue-600**/100**" dark-class="text-sky-400**/100**">Быстрая коричневая лиса...</p>
<p class="text-blue-600**/75**" dark-class="text-sky-400**/75**">Быстрая коричневая лиса...</p>
<p class="text-blue-600**/50**" dark-class="text-sky-400**/50**">Быстрая коричневая лиса...</p>
<p class="text-blue-600**/25**" dark-class="text-sky-400**/25**">Быстрая коричневая лиса...</p>
<p class="text-blue-600**/0**" dark-class="text-sky-400**/0**">Быстрая коричневая лиса...</p>
```

Вы можете использовать любое значение, определенное в вашей [шкале непрозрачности](/docs/opacity), или использовать произвольные значения, если вам нужно отклониться от маркеров вашего дизайна.

```html
<p class="text-blue-600**/[.06]**">Быстрая коричневая лиса...</p>
```
